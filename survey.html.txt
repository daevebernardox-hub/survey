    <script>
        // Floating No button
        const noButton = document.getElementById('no');
        noButton.addEventListener('mouseover', () => {
            const card = document.querySelector('.card');
            const cardRect = card.getBoundingClientRect();
            const buttonRect = noButton.getBoundingClientRect();
            
            // Random position within the card, avoiding edges
            const newLeft = Math.random() * (cardRect.width - buttonRect.width - 20) + 10;
            const newTop = Math.random() * (cardRect.height - buttonRect.height - 100) + 50; // Below the text
            
            noButton.style.left = `${newLeft}px`;
            noButton.style.top = `${newTop}px`;
        });

        // Yes button response
        document.getElementById('yes').addEventListener('click', () => {
            alert("Yay! Happy Valentine's Day, Cali! ❤️");
        });

        // Add floating hearts for ambiance
        const heartsContainer = document.getElementById('hearts');
        for (let i = 0; i < 10; i++) {
            const heart = document.createElement('div');
            heart.className = 'heart';
            heart.textContent = '❤️';
            heart.style.left = Math.random() * 100 + '%';
            heart.style.animationDelay = Math.random() * 3 + 's';
            heartsContainer.appendChild(heart);
        }
    </script>
